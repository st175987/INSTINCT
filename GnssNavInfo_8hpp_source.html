97 [[nodiscard]] std::shared_ptr<NAV::SatNavData> searchNavigationData(const SatId& satId, const InsTime& recvTime) const
98 {
99 if (!m_satellites.contains(satId)) { return nullptr; }
100
101 auto satNavData = m_satellites.at(satId).searchNavigationData(recvTime);
102 if (satNavData == nullptr)
103 {
104 [[maybe_unused]] auto printNavData = [&]() {
105 std::string ret;
106 for (const auto& navData : m_satellites.at(satId).getNavigationData())
107 {
108 ret += fmt::format("[{} - diff {:.0f}s], ", navData->refTime, std::abs((navData->refTime - recvTime).count()));
109 }
110 return ret.substr(0, ret.length() - 2);
111 };
112
113 LOG_TRACE("[{}][{}]: No navigation data found. Available data are at time: {}", satId, recvTime, printNavData());
114 }
115
116 return satNavData;
117 }
